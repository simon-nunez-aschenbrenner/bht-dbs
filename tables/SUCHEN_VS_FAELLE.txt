CREATE VIEW SUCHEN_VS_FAELLE (LAND, SUCHEN, INFIZIERTE, TOTE) AS SELECT NAME AS LAND, SUCHEN, INFIZIERTE, TOTE FROM LAND, ((SELECT LAND AS LAND_SUCHE, COUNT(DISTINCT ZEITPUNKT) AS SUCHEN FROM H5N1_SUCHE WHERE LAND IS NOT NULL GROUP BY LAND) FULL JOIN FAELLE ON LAND_SUCHE = FAELLE.LAND) WHERE NAME = LAND_SUCHE OR NAME = FAELLE.LAND ORDER BY 2
